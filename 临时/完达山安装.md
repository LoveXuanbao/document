## Redis

```bash
wget http://download.redis.io/releases/redis-7.0.13.tar.gz

tar xf redis-7.0.13.tar.gz

mv redis-7.0.13 /data1/

cd /data1/redis-7.0.13/

yum -y install gcc gcc-c++

make && make install

cd utils/

./install_server.sh


mkdir /data1/redis
mkdir /data1/redis/bin
mkdir /data1/redis/config
cd /src

cp mkreleasehdr.sh redis-benchmark redis-check-aof redis-check-rdb redis-cli redis-server redis-sentinel  /data1/redis/bin/

cp redis.conf /data1/redis/config/
cp sentinel.conf /data1/redis/config/

vim redis.conf
cd ../bin/
./redis-server /data1/redis/config/redis.conf

/data1/redis/bin/redis-sentinel /data1/redis/config/sentinel.conf
```
